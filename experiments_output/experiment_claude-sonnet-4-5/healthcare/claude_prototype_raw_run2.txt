<script>
let appointments = [
  {
    confirmNum: 'HC-2045',
    doctor: 'Dr. Smith',
    date: 'Oct 15, 2025',
    status: 'Confirmed',
    name: 'Sarah Johnson',
    type: 'General Checkup',
    time: '10:30 AM',
    notes: 'Arrive 15 minutes early'
  },
  {
    confirmNum: 'HC-1988',
    doctor: 'Dr. Lee',
    date: 'Oct 28, 2025',
    status: 'Pending',
    name: 'John Doe',
    type: 'Follow-up Visits',
    time: '2:00 PM',
    notes: 'Bring previous test results'
  }
];

let selectedAppointmentIndex = null;

function navigateTo(screenId) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
  
  document.getElementById(screenId).classList.add('active');
  document.querySelector(`[data-screen="${screenId}"]`).classList.add('active');
  
  window.scrollTo(0, 0);
}

document.querySelectorAll('.nav-link').forEach(link => {
  link.addEventListener('click', function(e) {
    e.preventDefault();
    navigateTo(this.dataset.screen);
  });
});

document.getElementById('appointmentForm').addEventListener('submit', function(e) {
  e.preventDefault();
  
  let isValid = true;
  document.querySelectorAll('.error-message').forEach(err => err.style.display = 'none');
  
  const name = document.getElementById('patientName').value.trim();
  const email = document.getElementById('email').value.trim();
  const phone = document.getElementById('phone').value.trim();
  const type = document.getElementById('appointmentType').value;
  const doctor = document.getElementById('doctor').value;
  const date = document.getElementById('appointmentDate').value;
  const time = document.getElementById('appointmentTime').value;
  
  if (!name) {
    document.getElementById('nameError').style.display = 'block';
    isValid = false;
  }
  if (!email || !email.includes('@')) {
    document.getElementById('emailError').style.display = 'block';
    isValid = false;
  }
  if (!phone) {
    document.getElementById('phoneError').style.display = 'block';
    isValid = false;
  }
  if (!type) {
    document.getElementById('typeError').style.display = 'block';
    isValid = false;
  }
  if (!doctor) {
    document.getElementById('doctorError').style.display = 'block';
    isValid = false;
  }
  if (!date) {
    document.getElementById('dateError').style.display = 'block';
    isValid = false;
  }
  if (!time) {
    document.getElementById('timeError').style.display = 'block';
    isValid = false;
  }
  
  if (!isValid) return;
  
  const confirmNum = 'HC-' + Math.floor(2000 + Math.random() * 1000);
  const dateObj = new Date(date);
  const formattedDate = dateObj.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  
  document.getElementById('confirmName').textContent = name;
  document.getElementById('confirmType').textContent = type;
  document.getElementById('confirmDoctor').textContent = doctor;
  document.getElementById('confirmDate').textContent = formattedDate;
  document.getElementById('confirmTime').textContent = time;
  document.getElementById('confirmNumber').textContent = confirmNum;
  
  const newAppointment = {
    confirmNum: confirmNum,
    doctor: doctor,
    date: formattedDate,
    status: 'Confirmed',
    name: name,
    type: type,
    time: time,
    notes: document.getElementById('notes').value || 'Arrive 15 minutes early'
  };
  
  appointments.unshift(newAppointment);
  updateAppointmentsList();
  
  navigateTo('confirmation');
});

function clearForm() {
  document.getElementById('appointmentForm').reset();
  document.querySelectorAll('.error-message').forEach(err => err.style.display = 'none');
}

function updateAppointmentsList() {
  const tbody = document.getElementById('appointmentsList');
  tbody.innerHTML = '';
  
  appointments.forEach((apt, index) => {
    const row = document.createElement('tr');
    row.onclick = () => selectAppointment(index);
    row.innerHTML = `
      <td>${apt.confirmNum}</td>
      <td>${apt.doctor}</td>
      <td>${apt.date}</td>
      <td><span class="status-badge status-${apt.status.toLowerCase()}">${apt.status}</span></td>
    `;
    tbody.appendChild(row);
  });
}

function selectAppointment(index) {
  selectedAppointmentIndex = index;
  const apt = appointments[index];
  
  document.querySelectorAll('.appointments-table tr').forEach(r => r.classList.remove('selected'));
  document.querySelectorAll('.appointments-table tr')[index + 1].classList.add('selected');
  
  document.getElementById('detailConfirmNum').textContent = apt.confirmNum;
  document.getElementById('detailNotes').textContent = apt.notes;
  document.getElementById('appointmentDetailsPanel').style.display = 'block';
}

function cancelAppointment() {
  if (selectedAppointmentIndex === null) {
    alert('Please select an appointment to cancel');
    return;
  }
  
  if (confirm('Are you sure you want to cancel this appointment?')) {
    appointments.splice(selectedAppointmentIndex, 1);
    updateAppointmentsList();
    document.getElementById('appointmentDetailsPanel').style.display = 'none';
    selectedAppointmentIndex = null;
    alert('Appointment cancelled successfully');
  }
}
</script>
