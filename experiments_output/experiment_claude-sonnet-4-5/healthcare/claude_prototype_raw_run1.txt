<script>
let appointments = [
  {
    confirmNum: 'HC-2045',
    doctor: 'Dr. Smith',
    date: 'Oct 15',
    status: 'Confirmed',
    location: 'Community Health Clinic',
    notes: 'Arrive 15 minutes early'
  },
  {
    confirmNum: 'HC-1988',
    doctor: 'Dr. Lee',
    date: 'Oct 28',
    status: 'Pending',
    location: 'Community Health Clinic',
    notes: 'Bring previous medical records'
  }
];

let selectedAppointmentIndex = 0;

function navigateTo(screenId) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById(screenId).classList.add('active');
  
  document.querySelectorAll('.nav-link').forEach(link => {
    link.classList.remove('active');
    if (link.dataset.screen === screenId) {
      link.classList.add('active');
    }
  });
  
  window.scrollTo(0, 0);
}

document.querySelectorAll('.nav-link').forEach(link => {
  link.addEventListener('click', function(e) {
    e.preventDefault();
    navigateTo(this.dataset.screen);
  });
});

document.getElementById('appointmentForm').addEventListener('submit', function(e) {
  e.preventDefault();
  
  document.querySelectorAll('.error-message').forEach(el => el.textContent = '');
  
  let isValid = true;
  
  const name = document.getElementById('patientName').value.trim();
  const email = document.getElementById('email').value.trim();
  const phone = document.getElementById('phone').value.trim();
  const type = document.getElementById('appointmentType').value;
  const doctor = document.getElementById('doctor').value;
  const date = document.getElementById('appointmentDate').value;
  const time = document.getElementById('appointmentTime').value;
  
  if (!name) {
    document.getElementById('nameError').textContent = 'Patient name is required';
    isValid = false;
  }
  
  if (!email) {
    document.getElementById('emailError').textContent = 'Email is required';
    isValid = false;
  }
  
  if (!phone) {
    document.getElementById('phoneError').textContent = 'Phone number is required';
    isValid = false;
  }
  
  if (!type) {
    document.getElementById('typeError').textContent = 'Please select appointment type';
    isValid = false;
  }
  
  if (!doctor) {
    document.getElementById('doctorError').textContent = 'Please select a doctor';
    isValid = false;
  }
  
  if (!date) {
    document.getElementById('dateError').textContent = 'Please select a date';
    isValid = false;
  }
  
  if (!time) {
    document.getElementById('timeError').textContent = 'Please select a time';
    isValid = false;
  }
  
  if (!isValid) return;
  
  const confirmNum = 'HC-' + Math.floor(2000 + Math.random() * 1000);
  
  const dateObj = new Date(date);
  const formattedDate = dateObj.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  
  document.getElementById('confirmName').textContent = name;
  document.getElementById('confirmType').textContent = type;
  document.getElementById('confirmDoctor').textContent = doctor;
  document.getElementById('confirmDate').textContent = formattedDate;
  document.getElementById('confirmTime').textContent = time;
  document.getElementById('confirmNumber').textContent = confirmNum;
  
  const notes = document.getElementById('notes').value.trim() || 'None';
  
  appointments.unshift({
    confirmNum: confirmNum,
    doctor: doctor,
    date: dateObj.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }),
    status: 'Confirmed',
    location: 'Community Health Clinic',
    notes: notes
  });
  
  updateAppointmentsList();
  
  navigateTo('confirmation');
});

function clearForm() {
  document.getElementById('appointmentForm').reset();
  document.querySelectorAll('.error-message').forEach(el => el.textContent = '');
}

function updateAppointmentsList() {
  const tbody = document.getElementById('appointmentsList');
  tbody.innerHTML = '';
  
  appointments.forEach((appt, index) => {
    const row = document.createElement('tr');
    row.onclick = () => selectAppointment(index);
    row.innerHTML = `
      <td>${appt.confirmNum}</td>
      <td>${appt.doctor}</td>
      <td>${appt.date}</td>
      <td><span class="status-${appt.status.toLowerCase()}">${appt.status}</span></td>
    `;
    tbody.appendChild(row);
  });
}

function selectAppointment(index) {
  selectedAppointmentIndex = index;
  const appt = appointments[index];
  
  document.getElementById('selectedConfirm').textContent = appt.confirmNum;
  document.getElementById('selectedNotes').textContent = appt.notes;
}

function cancelAppointment() {
  if (confirm('Are you sure you want to cancel this appointment?')) {
    appointments.splice(selectedAppointmentIndex, 1);
    updateAppointmentsList();
    
    if (appointments.length > 0) {
      selectAppointment(0);
    } else {
      document.getElementById('appointmentDetails').style.display = 'none';
    }
  }
}

updateAppointmentsList();
</script>
