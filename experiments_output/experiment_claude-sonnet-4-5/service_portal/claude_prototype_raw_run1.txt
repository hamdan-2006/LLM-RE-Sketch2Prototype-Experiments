// Navigation
function navigateTo(screenId) {
  document.querySelectorAll('.screen').forEach(screen => {
    screen.classList.remove('active');
  });
  document.getElementById(screenId).classList.add('active');
  
  document.querySelectorAll('.nav-link').forEach(link => {
    link.classList.remove('active');
  });
  document.querySelector(`[data-screen="${screenId}"]`)?.classList.add('active');
  
  window.scrollTo(0, 0);
}

document.querySelectorAll('.nav-link').forEach(link => {
  link.addEventListener('click', (e) => {
    e.preventDefault();
    const screen = e.target.getAttribute('data-screen');
    navigateTo(screen);
  });
});

// Form submission
document.getElementById('submitForm').addEventListener('submit', (e) => {
  e.preventDefault();
  
  const name = document.getElementById('name').value.trim();
  const email = document.getElementById('email').value.trim();
  const category = document.getElementById('category').value;
  const description = document.getElementById('description').value.trim();
  
  if (!name || !email || !category || !description) {
    const errorDiv = document.getElementById('submitError');
    errorDiv.textContent = 'Please fill in all required fields.';
    errorDiv.classList.add('show');
    return;
  }
  
  // Generate ticket ID
  const ticketId = 'IT-' + Math.floor(Math.random() * 9000 + 1000);
  
  // Get current date
  const now = new Date();
  const dateStr = now.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  
  // Update confirmation screen
  document.getElementById('confirmTicketId').textContent = ticketId;
  document.getElementById('confirmCategory').textContent = category;
  document.getElementById('confirmSubmitter').textContent = name;
  document.getElementById('confirmDate').textContent = dateStr;
  
  // Store ticket data for tracking
  const ticketData = {
    id: ticketId,
    category: category,
    submitter: name,
    status: 'Open',
    date: dateStr,
    updated: dateStr,
    notes: 'Ticket received and awaiting assignment'
  };
  
  sessionStorage.setItem('lastTicket', JSON.stringify(ticketData));
  
  navigateTo('confirmation');
});

// Clear form
function clearForm() {
  document.getElementById('submitForm').reset();
  document.getElementById('submitError').classList.remove('show');
}

// Track ticket
document.getElementById('searchForm').addEventListener('submit', (e) => {
  e.preventDefault();
  
  const ticketId = document.getElementById('ticketId').value.trim();
  
  if (!ticketId) {
    return;
  }
  
  // Check if it's the last submitted ticket
  const lastTicket = sessionStorage.getItem('lastTicket');
  if (lastTicket) {
    const ticket = JSON.parse(lastTicket);
    if (ticketId === ticket.id || ticketId === '2025-164' || ticketId.toLowerCase().includes('it-')) {
      document.getElementById('trackTicketId').textContent = ticket.id || '2025-164';
      document.getElementById('trackCategory').textContent = ticket.category || 'Network Support';
      document.getElementById('trackSubmitter').textContent = ticket.submitter || 'Student';
      document.getElementById('trackUpdated').textContent = ticket.updated || 'April 29, 2026';
      document.getElementById('trackNotes').textContent = ticket.notes || 'Issue currently being investigated';
      
      const statusBadge = document.getElementById('trackStatus');
      statusBadge.textContent = ticket.status === 'Open' ? 'In Progress' : ticket.status;
      statusBadge.className = 'status-badge status-in-progress';
      
      document.getElementById('ticketResults').style.display = 'block';
      return;
    }
  }
  
  // Default ticket for demo
  document.getElementById('trackTicketId').textContent = '2025-164';
  document.getElementById('trackCategory').textContent = 'Network Support';
  document.getElementById('trackSubmitter').textContent = 'Student';
  document.getElementById('trackUpdated').textContent = 'April 29, 2026';
  document.getElementById('trackNotes').textContent = 'Issue currently being investigated';
  
  const statusBadge = document.getElementById('trackStatus');
  statusBadge.textContent = 'In Progress';
  statusBadge.className = 'status-badge status-in-progress';
  
  document.getElementById('ticketResults').style.display = 'block';
});
</script>
