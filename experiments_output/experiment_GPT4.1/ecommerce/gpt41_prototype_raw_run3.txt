      { id: 8, name: "Noise Cancelling Headphones", category: "Headphones", price: 149, featured: false },
      { id: 9, name: "Gaming Mouse", category: "Games", price: 45, featured: false },
      { id: 10, name: "Ultrabook", category: "Laptops", price: 999, featured: false }
    ];
    const homeFeaturedIds = [1, 3, 7];
    const slideshowItems = [
      { title: "Laptop", desc: "High performance laptop for work and play", img: "💻" },
      { title: "Wireless Mouse", desc: "Precision and comfort for gaming", img: "🖱️" },
      { title: "Headphones", desc: "Crystal clear sound and comfort", img: "🎧" },
      { title: "Smart Watch", desc: "Stay connected on the go", img: "⌚" }
    ];
    // --- State ---
    let cart = [];
    let filteredProducts = [...allProducts];
    let searchTerm = "";
    let slideshowIdx = 0;
    let lastOrder = null;
    // --- Utility ---
    function $(id) { return document.getElementById(id); }
    function formatPrice(p) { return "$" + p.toFixed(2); }
    function getCartCount() {
      return cart.reduce((sum, item) => sum + item.qty, 0);
    }
    function getCartTotals() {
      let subtotal = cart.reduce((sum, item) => sum + item.price * item.qty, 0);
      let tax = Math.round(subtotal * 0.08 * 100) / 100;
      let total = Math.round((subtotal + tax) * 100) / 100;
      return { subtotal, tax, total };
    }
    function updateCartCount() {
      const count = getCartCount();
      $("cart-count").textContent = count > 0 ? `(${count})` : "";
    }
    // --- Navigation ---
    function showScreen(screen) {
      const screens = ["home", "products", "cart", "checkout", "confirmation"];
      screens.forEach(s => {
        $("screen-" + s).style.display = (s === screen) ? "" : "none";
        $("nav-" + s)?.classList.remove("active");
      });
      $("nav-" + screen)?.classList.add("active");
      if (screen === "home") renderHome();
      if (screen === "products") renderProducts();
      if (screen === "cart") renderCart();
      if (screen === "checkout") renderCheckout();
      if (screen === "confirmation") renderConfirmation();
      window.scrollTo(0, 0);
    }
    // --- Home Page ---
    function renderHome() {
      // Featured products
      const featured = allProducts.filter(p => homeFeaturedIds.includes(p.id));
      $("home-featured-products").innerHTML = featured.map(p => `
        <div class="product-card">
          <div class="img-placeholder">${getProductIcon(p.category)}</div>
          <div class="name">${p.name}</div>
          <div class="price">${formatPrice(p.price)}</div>
          <button class="btn" onclick="addToCart(${p.id})">Add to Cart</button>
        </div>
      `).join("");
    }
    function filterProducts(category) {
      filteredProducts = allProducts.filter(p => p.category === category);
      searchTerm = "";
      $("search-products").value = "";
      showScreen('products');
      renderProducts();
    }
    function showDeals() {
      filteredProducts = allProducts.filter(p => [1,3,7].includes(p.id));
      searchTerm = "";
      $("search-products").value = "";
      showScreen('products');
      renderProducts();
    }
    // --- Products Page ---
    function renderProducts() {
      // Slideshow
      renderSlideshow();
      // Products list
      let products = filteredProducts;
      if (searchTerm) {
        products = products.filter(p => p.name.toLowerCase().includes(searchTerm.toLowerCase()));
      }
      $("products-list").innerHTML = products.length === 0
        ? `<div style="color:#888;padding:1rem;">No products found.</div>`
        : products.map(p => `
          <div class="product-row">
            <span class="name">${p.name}</span>
            <span class="price">${formatPrice(p.price)}</span>
            <button class="btn" onclick="addToCart(${p.id})">Add to Cart</button>
          </div>
        `).join("");
      // Reset filter
      filteredProducts = allProducts;
    }
    function searchProducts() {
      searchTerm = $("search-products").value.trim();
      renderProducts();
    }
    // --- Slideshow ---
    function renderSlideshow() {
      const item = slideshowItems[slideshowIdx];
      $("slideshow-content").innerHTML = `
        <span style="font-size:2rem;margin-right:0.7rem;">${item.img}</span>
        <span>
          <strong>${item.title}</strong><br>
          <span style="font-size:1rem;color:#444;">${item.desc}</span>
        </span>
      `;
    }
    function prevSlide() {
      slideshowIdx = (slideshowIdx - 1 + slideshowItems.length) % slideshowItems.length;
      renderSlideshow();
    }
    function nextSlide() {
      slideshowIdx = (slideshowIdx + 1) % slideshowItems.length;
      renderSlideshow();
    }
    // --- Cart ---
    function addToCart(id) {
      const prod = allProducts.find(p => p.id === id);
      if (!prod) return;
      let item = cart.find(i => i.id === id);
      if (item) {
        item.qty += 1;
      } else {
        cart.push({ id: prod.id, name: prod.name, price: prod.price, qty: 1 });
      }
      updateCartCount();
      renderCart();
      // Optional: show a quick feedback
      showScreen('cart');
    }
    function renderCart() {
      updateCartCount();
      const tbody = $("cart-items");
      if (cart.length === 0) {
        tbody.innerHTML = `<tr><td colspan="4" style="color:#888;text-align:center;padding:1.5rem;">Your cart is empty.</td></tr>`;
        $("proceed-checkout-btn").disabled = true;
      } else {
        $("proceed-checkout-btn").disabled = false;
        tbody.innerHTML = cart.map(item => `
          <tr>
            <td>${item.name}</td>
            <td>${item.qty}</td>
            <td>${formatPrice(item.price * item.qty)}</td>
            <td class="action"><button class="btn danger" onclick="removeFromCart(${item.id})">Remove</button></td>
          </tr>
        `).join("");
      }
      const totals = getCartTotals();
      $("cart-subtotal").textContent = formatPrice(totals.subtotal);
      $("cart-tax").textContent = formatPrice(totals.tax);
      $("cart-total").textContent = formatPrice(totals.total);
    }
    function removeFromCart(id) {
      cart = cart.filter(item => item.id !== id);
      updateCartCount();
      renderCart();
    }
    function continueShopping() {
      showScreen('products');
    }
    function proceedToCheckout() {
      if (cart.length === 0) return;
      showScreen('checkout');
    }
    // --- Checkout ---
    function renderCheckout() {
      const totals = getCartTotals();
      $("checkout-total").textContent = formatPrice(totals.total);
      $("checkout-error").style.display = "none";
      $("checkout-form").reset();
    }
    function cancelCheckout() {
      showScreen('cart');
    }
    function placeOrder() {
      // Validate
      const name = $("checkout-name").value.trim();
      const address = $("checkout-address").value.trim();
      const city = $("checkout-city").value.trim();
      const state = $("checkout-state").value.trim();
      const zip = $("checkout-zip").value.trim();
      const payment = $("checkout-payment").value;
      const card = $("checkout-card").value.trim();
      const exp = $("checkout-exp").value.trim();
      let error = "";
      if (!name || !address || !city || !state || !zip || !payment || !card || !exp) {
        error = "Please fill in all required fields.";
      } else if (!/^\d{2}\/\d{2}$/.test(exp)) {
        error = "Expiration date must be in MM/YY format.";
      } else if (!/^\d{4,}$/.test(card.replace(/\s/g,''))) {
        error = "Invalid card number.";
      }
      if (error) {
        $("checkout-error").textContent = error;
        $("checkout-error").style.display = "";
        return;
      }
      // Place order
      const totals = getCartTotals();
      lastOrder = {
        orderNo: "ORD-" + Math.floor(1000 + Math.random()*9000),
        items: cart.map(i => ({ name: i.name, qty: i.qty, price: i.price })),
        total: totals.total,
        paymentStatus: "APPROVED",
        delivery: getEstimatedDelivery(),
        shipping: { name, address, city, state, zip }
      };
      cart = [];
      updateCartCount();
      showScreen('confirmation');
    }
    function getEstimatedDelivery() {
      const d = new Date();
      d.setDate(d.getDate() + 5);
      return d.toLocaleDateString(undefined, { month: 'short', day: 'numeric', year: '2-digit' });
    }
    // --- Order Confirmation ---
    function renderConfirmation() {
      if (!lastOrder) return;
      const o = lastOrder;
      $("order-summary").innerHTML = `
        <div class="order-summary-row"><strong>Order #:</strong> ${o.orderNo}</div>
        <div class="order-summary-row"><strong>Items:</strong> ${o.items.map(i => `${i.name} x${i.qty}`).join(', ')}</div>
        <div class="order-summary-row"><strong>Total Paid:</strong> ${formatPrice(o.total)}</div>
        <div class="order-summary-row"><strong>Payment Status:</strong> ${o.paymentStatus}</div>
        <div class="order-summary-row"><strong>Estimated Delivery:</strong> ${o.delivery}</div>
        <div class="order-summary-row"><strong>Shipping Address:</strong><br>
          ${o.shipping.name}<br>
          ${o.shipping.address}<br>
          ${o.shipping.city}, ${o.shipping.state} ${o.shipping.zip}
        </div>
      `;
    }
    // --- Helpers ---
    function getProductIcon(category) {
      switch (category) {
        case "Laptops": return "💻";
        case "Headphones": return "🎧";
        case "Games": return "🎮";
        default: return "📦";
      }
    }

    renderHome();

    window.showScreen = showScreen;
