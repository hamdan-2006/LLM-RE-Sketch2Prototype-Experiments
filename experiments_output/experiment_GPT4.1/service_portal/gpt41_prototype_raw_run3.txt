  <script>
    // --- Navigation logic ---
    const screens = {
      home: document.getElementById('screen-home'),
      submit: document.getElementById('screen-submit'),
      confirm: document.getElementById('screen-confirm'),
      track: document.getElementById('screen-track'),
      help: document.getElementById('screen-help')
    };
    const navLinks = {
      home: document.getElementById('nav-home'),
      submit: document.getElementById('nav-submit'),
      track: document.getElementById('nav-track'),
      help: document.getElementById('nav-help')
    };
    function showScreen(screen) {
      for (const key in screens) {
        screens[key].classList.toggle('hide', key !== screen);
      }
      for (const key in navLinks) {
        navLinks[key].classList.toggle('active', key === screen);
      }
      // Focus first input if present
      if (screen === 'submit') setTimeout(() => document.getElementById('input-name').focus(), 100);
      if (screen === 'track') setTimeout(() => document.getElementById('input-ticketid').focus(), 100);
      window.scrollTo(0,0);
    }
    navLinks.home.onclick = () => showScreen('home');
    navLinks.submit.onclick = () => showScreen('submit');
    navLinks.track.onclick = () => showScreen('track');
    navLinks.help.onclick = () => showScreen('help');
    // Home "Submit New Request" button
    document.getElementById('btn-new-request').onclick = () => showScreen('submit');
    // Confirmation screen buttons
    document.getElementById('btn-track-from-confirm').onclick = () => showScreen('track');
    document.getElementById('btn-return-home-from-confirm').onclick = () => showScreen('home');
    // Help screen return
    document.getElementById('btn-return-home-from-help').onclick = () => showScreen('home');
    // Track screen return
    document.getElementById('btn-return-home-from-track').onclick = () => showScreen('home');

    // --- Submit Support Request Form Logic ---
    const supportForm = document.getElementById('support-form');
    const errName = document.getElementById('err-name');
    const errEmail = document.getElementById('err-email');
    const errCategory = document.getElementById('err-category');
    const errDesc = document.getElementById('err-desc');
    let lastTicket = null;
    function validateEmail(email) {
      return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    }
    supportForm.onsubmit = function(e) {
      e.preventDefault();
      // Hide all errors
      [errName, errEmail, errCategory, errDesc].forEach(e=>e.classList.add('hide'));
      let valid = true;
      const name = supportForm.name.value.trim();
      const email = supportForm.email.value.trim();
      const category = supportForm.category.value;
      const desc = supportForm.desc.value.trim();
      if (!name) { errName.classList.remove('hide'); valid = false; }
      if (!email || !validateEmail(email)) { errEmail.classList.remove('hide'); valid = false; }
      if (!category) { errCategory.classList.remove('hide'); valid = false; }
      if (!desc) { errDesc.classList.remove('hide'); valid = false; }
      if (!valid) return;
      // Generate ticket
      const ticketId = 'IT-' + (2000 + Math.floor(Math.random()*9000));
      const now = new Date();
      const dateStr = now.toLocaleDateString('en-US', {year:'numeric', month:'short', day:'numeric'});
      lastTicket = {
        id: ticketId,
        category,
        submitter: name,
        email,
        status: 'Open',
        date: dateStr,
        response: '24 hours'
      };
      // Fill confirmation screen
      const summary = document.getElementById('ticket-summary');
      summary.innerHTML = `
        <div class="ticket-summary-row"><span class="ticket-summary-label">Ticket ID:</span> ${ticketId}</div>
        <div class="ticket-summary-row"><span class="ticket-summary-label">Category:</span> ${category}</div>
        <div class="ticket-summary-row"><span class="ticket-summary-label">Submitted by:</span> ${name}</div>
        <div class="ticket-summary-row"><span class="ticket-summary-label">Status:</span> <span class="status-badge status-open">Open</span></div>
        <div class="ticket-summary-row"><span class="ticket-summary-label">Submitted Date:</span> ${dateStr}</div>
        <div class="ticket-summary-row"><span class="ticket-summary-label">Estimated Response Time:</span> 24 hours</div>
      `;
      showScreen('confirm');
      supportForm.reset();
    };
    // Clear form button
    document.getElementById('btn-clear-form').onclick = function() {
      supportForm.reset();
      [errName, errEmail, errCategory, errDesc].forEach(e=>e.classList.add('hide'));
      document.getElementById('input-name').focus();
    };

    // --- Track Request Logic ---
    const trackForm = document.getElementById('track-form');
    const errTicketId = document.getElementById('err-ticketid');
    const ticketInfo = document.getElementById('ticket-info');
    // Sample tickets for demo
    const sampleTickets = {
      'IT-2045': {
        id: 'IT-2045',
        category: 'Network Support',
        status: 'Open',
        submitter: 'Student',
        lastUpdated: 'Oct 10, 2025',
        notes: 'Awaiting technician assignment.'
      },
      '2055-104': {
        id: '2055-104',
        category: 'Network Support',
        status: 'In Progress',
        submitter: 'Student',
        lastUpdated: 'April 29, 2026',
        notes: 'Issue currently being investigated.'
      },
      'IT-1001': {
        id: 'IT-1001',
        category: 'Password Reset',
        status: 'Closed',
        submitter: 'Faculty',
        lastUpdated: 'Mar 2, 2024',
        notes: 'Password reset instructions sent.'
      }
    };
    function renderTicketInfo(ticket) {
      if (!ticket) {
        ticketInfo.innerHTML = `<div style="color:#888;">No ticket found for this ID.</div>`;
        return;
      }
      let statusClass = 'status-open';
      if (ticket.status.toLowerCase().includes('progress')) statusClass = 'status-inprogress';
      if (ticket.status.toLowerCase().includes('closed')) statusClass = 'status-closed';
      ticketInfo.innerHTML = `
        <div class="ticket-info-row"><span class="ticket-info-label">Ticket #:</span> ${ticket.id}</div>
        <div class="ticket-info-row"><span class="ticket-info-label">Category:</span> ${ticket.category}</div>
        <div class="ticket-info-row"><span class="ticket-info-label">Status:</span> <span class="status-badge ${statusClass}">${ticket.status}</span></div>
        <div class="ticket-info-row"><span class="ticket-info-label">Submitted by:</span> ${ticket.submitter}</div>
        <div class="ticket-info-row"><span class="ticket-info-label">Last Updated:</span> ${ticket.lastUpdated}</div>
        <div class="ticket-info-row"><span class="ticket-info-label">Technician Notes:</span> ${ticket.notes}</div>
      `;
    }
    trackForm.onsubmit = function(e) {
      e.preventDefault();
      errTicketId.classList.add('hide');
      const id = trackForm.ticketid.value.trim();
      if (!id) {
        errTicketId.textContent = 'Please enter a valid Ticket ID.';
        errTicketId.classList.remove('hide');
        ticketInfo.innerHTML = `<div style="color:#888;">No ticket selected.</div>`;
        return;
      }
      // Check sample tickets and last submitted ticket
      let ticket = sampleTickets[id];
      if (!ticket && lastTicket && (id === lastTicket.id)) {
        ticket = {
          id: lastTicket.id,
          category: lastTicket.category,
          status: lastTicket.status,
          submitter: lastTicket.submitter,
          lastUpdated: lastTicket.date,
          notes: 'Awaiting technician assignment.'
        };
      }
      renderTicketInfo(ticket);
    };

    // Autofill ticket info if navigated from confirmation
    document.getElementById('btn-track-from-confirm').onclick = function() {
      showScreen('track');
      setTimeout(() => {
        if (lastTicket) {
          document.getElementById('input-ticketid').value = lastTicket.id;
          renderTicketInfo({
            id: lastTicket.id,
            category: lastTicket.category,
            status: lastTicket.status,
            submitter: lastTicket.submitter,
            lastUpdated: lastTicket.date,
            notes: 'Awaiting technician assignment.'
          });
        }
      }, 120);
    };

    // Keyboard navigation: Enter on nav links
    for (const key in navLinks) {
      navLinks[key].onkeydown = function(e) {
        if (e.key === 'Enter' || e.key === ' ') this.click();
      };
    }
    // Set initial screen
    showScreen('home');
  </script>
