<script>
  // --- Data ---
  const SERVICES = [
    "General checkup",
    "Pediatric care",
    "Vaccination services",
    "Follow-up visits"
  ];
  const ANNOUNCEMENTS = [
    "Flu shots available this week",
    "New patient forms updated"
  ];
  const APPOINTMENT_TYPES = [
    "General checkup",
    "Pediatric care",
    "Vaccination",
    "Follow-up visit"
  ];
  const DOCTORS = [
    { name: "Dr. Smith", types: ["General checkup", "Vaccination", "Follow-up visit"] },
    { name: "Dr. Lee", types: ["Pediatric care", "Vaccination"] },
    { name: "Dr. Patel", types: ["General checkup", "Follow-up visit"] }
  ];
  const APPT_TIMES = [
    "09:00 AM", "09:30 AM", "10:00 AM", "10:30 AM",
    "11:00 AM", "11:30 AM", "01:00 PM", "01:30 PM",
    "02:00 PM", "02:30 PM", "03:00 PM", "03:30 PM"
  ];
  const FAQS = [
    {
      q: "How do I schedule an appointment?",
      a: 'Select "Schedule Appointment" from the menu and fill out the form.'
    },
    {
      q: "Can I cancel an appointment online?",
      a: 'Yes, use the My Appointments page to cancel upcoming appointments.'
    },
    {
      q: "How will I receive confirmation?",
      a: "A confirmation appears immediately after booking with your details."
    }
  ];
  const SUPPORT_EMAIL = "support@clinic.org";
  const SUPPORT_PHONE = "(000) 555-0010";

  // --- State ---
  let appointments = [
    {
      confirmation: "HC-2045",
      patient: "Sarah Johnson",
      type: "General checkup",
      doctor: "Dr. Smith",
      date: "2025-10-15",
      time: "10:30 AM",
      status: "confirmed",
      location: "Community Health Clinic",
      notes: "Arrive 15 minutes early"
    },
    {
      confirmation: "HC-1988",
      patient: "Alex Kim",
      type: "Pediatric care",
      doctor: "Dr. Lee",
      date: "2025-10-28",
      time: "09:00 AM",
      status: "pending",
      location: "Community Health Clinic",
      notes: "Arrive 15 minutes early"
    }
  ];
  let selectedScreen = "home";
  let selectedAppointmentIdx = 0;
  let lastScheduledAppointment = null;

  // --- Utility Functions ---
  function formatDate(dateStr) {
    if (!dateStr) return "";
    const d = new Date(dateStr);
    if (isNaN(d)) return dateStr;
    const opts = { year: 'numeric', month: 'short', day: 'numeric' };
    return d.toLocaleDateString(undefined, opts);
  }
  function generateConfirmationNumber() {
    return "HC-" + (Math.floor(Math.random() * 9000) + 1000);
  }
  function getDoctorsForType(type) {
    return DOCTORS.filter(doc => doc.types.includes(type));
  }
  function getTimesForDoctorAndDate(doctor, date) {
    // For prototype: always return all times
    return APPT_TIMES;
  }
  function getStatusLabel(status) {
    if (status === "confirmed") return '<span class="appt-status confirmed">Confirmed</span>';
    if (status === "pending") return '<span class="appt-status pending">Pending</span>';
    return status;
  }

  // --- Navigation ---
  function setScreen(screen, opts={}) {
    selectedScreen = screen;
    // Update nav active state
    document.querySelectorAll('.nav-link').forEach(btn => {
      btn.classList.toggle('active', btn.dataset.screen === screen);
    });
    // Render screen
    switch(screen) {
      case "home": renderHome(); break;
      case "schedule": renderScheduleForm(); break;
      case "confirmation": renderConfirmation(); break;
      case "appointments": renderAppointments(opts.selectedIdx); break;
      case "help": renderHelp(); break;
    }
    window.scrollTo(0,0);
  }

  // --- Screens ---
  function renderHome() {
    document.querySelector('main').innerHTML = `
      <div class="center">
        <h2 style="margin-bottom:0.2rem;">Community Health Clinic Portal</h2>
        <div class="logo-placeholder" aria-label="Clinic Logo">CHC</div>
        <div style="margin-bottom:1.2rem;color:var(--muted);font-size:1.08rem;">
          Welcome to Community Health Clinic
        </div>
      </div>
      <div class="section center">
        <button class="btn" id="btn-schedule-home">Schedule New Appointment</button>
      </div>
      <div class="section">
        <div class="section-title">Services Available:</div>
        <ul class="services-list">
          ${SERVICES.map(s => `<li>${s}</li>`).join("")}
        </ul>
      </div>
      <div class="section">
        <div class="section-title">Announcements</div>
        <ul class="announcements-list">
          ${ANNOUNCEMENTS.map(a => `<li>${a}</li>`).join("")}
        </ul>
      </div>
    `;
    document.getElementById('btn-schedule-home').onclick = () => setScreen('schedule');
  }

  function renderScheduleForm() {
    // Defaults
    let today = new Date();
    let minDate = today.toISOString().split('T')[0];
    let maxDate = new Date(today.getFullYear(), today.getMonth()+2, today.getDate()).toISOString().split('T')[0];
    // Form state
    let form = {
      patient: "",
      email: "",
      phone: "",
      type: APPOINTMENT_TYPES[0],
      doctor: getDoctorsForType(APPOINTMENT_TYPES[0])[0]?.name || "",
      date: minDate,
      time: APPT_TIMES[0],
      notes: ""
    };
    let errors = {};
    // Render
    function updateForm() {
      // Update doctor options
      let doctorOptions = getDoctorsForType(form.type).map(doc =>
        `<option value="${doc.name}" ${form.doctor===doc.name?'selected':''}>${doc.name}</option>`
      ).join("");
      // Update time options
      let timeOptions = getTimesForDoctorAndDate(form.doctor, form.date).map(t =>
        `<option value="${t}" ${form.time===t?'selected':''}>${t}</option>`
      ).join("");
      // Error messages
      let errorHtml = Object.values(errors).length
        ? `<div class="error-msg">${Object.values(errors).join('<br>')}</div>` : "";
      document.querySelector('main').innerHTML = `
        <h2 class="center" style="margin-bottom:1.2rem;">Schedule Appointment</h2>
        <form id="appt-form" autocomplete="off" novalidate>
          ${errorHtml}
          <div class="form-group">
            <label for="patient">Patient Name *</label>
            <input type="text" id="patient" name="patient" value="${form.patient}" maxlength="40" required>
          </div>
          <div class="form-row">
            <div class="form-group">
              <label for="email">Email Address *</label>
              <input type="email" id="email" name="email" value="${form.email}" maxlength="40" required>
            </div>
            <div class="form-group">
              <label for="phone">Phone Number *</label>
              <input type="tel" id="phone" name="phone" value="${form.phone}" maxlength="20" required>
            </div>
          </div>
          <div class="form-row">
            <div class="form-group">
              <label for="type">Appointment Type *</label>
              <select id="type" name="type" required>
                ${APPOINTMENT_TYPES.map(t =>
                  `<option value="${t}" ${form.type===t?'selected':''}>${t}</option>`
                ).join("")}
              </select>
            </div>
            <div class="form-group">
              <label for="doctor">Preferred Doctor *</label>
              <select id="doctor" name="doctor" required>
                ${doctorOptions}
              </select>
            </div>
          </div>
          <div class="form-row">
            <div class="form-group">
              <label for="date">Appointment Date *</label>
              <input type="date" id="date" name="date" value="${form.date}" min="${minDate}" max="${maxDate}" required>
            </div>
            <div class="form-group">
              <label for="time">Appointment Time *</label>
              <select id="time" name="time" required>
                ${timeOptions}
              </select>
            </div>
          </div>
          <div class="form-group">
            <label for="notes">Additional Notes</label>
            <textarea id="notes" name="notes" rows="2" maxlength="120">${form.notes}</textarea>
          </div>
          <div class="form-actions">
            <button type="submit" class="btn">Confirm Appointment</button>
            <button type="button" class="btn secondary" id="clear-form">Clear Form</button>
          </div>
        </form>
      `;
      // Event handlers
      let $form = document.getElementById('appt-form');
      $form.patient.oninput = e => { form.patient = e.target.value; };
      $form.email.oninput = e => { form.email = e.target.value; };
      $form.phone.oninput = e => { form.phone = e.target.value; };
      $form.type.onchange = e => {
        form.type = e.target.value;
        let docs = getDoctorsForType(form.type);
        form.doctor = docs[0]?.name || "";
        updateForm();
      };
      $form.doctor.onchange = e => {
        form.doctor = e.target.value;
        form.time = getTimesForDoctorAndDate(form.doctor, form.date)[0];
        updateForm();
      };
      $form.date.onchange = e => {
        form.date = e.target.value;
        form.time = getTimesForDoctorAndDate(form.doctor, form.date)[0];
        updateForm();
      };
      $form.time.onchange = e => { form.time = e.target.value; };
      $form.notes.oninput = e => { form.notes = e.target.value; };
      $form.onsubmit = function(ev) {
        ev.preventDefault();
        errors = {};
        // Validation
        if (!form.patient.trim()) errors.patient = "Patient name is required.";
        if (!form.email.trim() || !/^[^@]+@[^@]+\.[^@]+$/.test(form.email)) errors.email = "Valid email is required.";
        if (!form.phone.trim() || !/^[\d\-\+\(\) ]{7,}$/.test(form.phone)) errors.phone = "Valid phone number is required.";
        if (!form.type) errors.type = "Appointment type is required.";
        if (!form.doctor) errors.doctor = "Preferred doctor is required.";
        if (!form.date) errors.date = "Appointment date is required.";
        if (!form.time) errors.time = "Appointment time is required.";
        if (Object.keys(errors).length) {
          updateForm();
          return;
        }
        // Save appointment
        let confirmation = generateConfirmationNumber();
        let appt = {
          confirmation,
          patient: form.patient.trim(),
          type: form.type,
          doctor: form.doctor,
          date: form.date,
          time: form.time,
          status: "confirmed",
          location: "Community Health Clinic",
          notes: form.notes.trim() || "Arrive 15 minutes early"
        };
        appointments.unshift(appt);
        lastScheduledAppointment = appt;
        setScreen('confirmation');
      };
      document.getElementById('clear-form').onclick = function() {
        form = {
          patient: "",
          email: "",
          phone: "",
          type: APPOINTMENT_TYPES[0],
          doctor: getDoctorsForType(APPOINTMENT_TYPES[0])[0]?.name || "",
          date: minDate,
          time: APPT_TIMES[0],
          notes: ""
        };
        errors = {};
        updateForm();
      };
    }
    updateForm();
  }

  function renderConfirmation() {
    let appt = lastScheduledAppointment || appointments[0];
    document.querySelector('main').innerHTML = `
      <h2 class="center" style="margin-bottom:1.2rem;">Appointment Confirmation</h2>
      <div class="success-msg center">Your appointment has been scheduled successfully.</div>
      <div class="appt-summary">
        <div class="appt-summary-row"><span class="appt-summary-label">Patient Name:</span> ${appt.patient}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Appointment Type:</span> ${appt.type}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Doctor:</span> ${appt.doctor}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Date:</span> ${formatDate(appt.date)}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Time:</span> ${appt.time}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Confirmation #:</span> ${appt.confirmation}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Status:</span> ${getStatusLabel(appt.status)}</div>
      </div>
      <div class="form-actions center" style="justify-content:center;">
        <button class="btn" id="btn-view-appts">View My Appointments</button>
        <button class="btn secondary" id="btn-return-home">Return Home</button>
      </div>
    `;
    document.getElementById('btn-view-appts').onclick = () => setScreen('appointments', {selectedIdx: 0});
    document.getElementById('btn-return-home').onclick = () => setScreen('home');
  }

  function renderAppointments(selectedIdx=0) {
    if (appointments.length === 0) {
      document.querySelector('main').innerHTML = `
        <h2 class="center" style="margin-bottom:1.2rem;">My Appointments</h2>
        <div class="center" style="margin:2rem 0 2.5rem 0;">No appointments found.</div>
        <div class="form-actions center" style="justify-content:center;">
          <button class="btn secondary" id="btn-return-home">Return Home</button>
        </div>
      `;
      document.getElementById('btn-return-home').onclick = () => setScreen('home');
      return;
    }
    selectedAppointmentIdx = typeof selectedIdx === "number" ? selectedIdx : 0;
    let appt = appointments[selectedAppointmentIdx];
    document.querySelector('main').innerHTML = `
      <h2 class="center" style="margin-bottom:1.2rem;">My Appointments</h2>
      <div class="section-title">Upcoming Appointments</div>
      <div style="overflow-x:auto;">
        <table>
          <thead>
            <tr>
              <th>Confirmation #</th>
              <th>Doctor</th>
              <th>Date</th>
              <th>Status</th>
            </tr>
          </thead>
          <tbody>
            ${appointments.map((a,i) => `
              <tr class="${i===selectedAppointmentIdx?'selected':''}" data-idx="${i}" tabindex="0" style="cursor:pointer;">
                <td>${a.confirmation}</td>
                <td>${a.doctor}</td>
                <td>${formatDate(a.date)}</td>
                <td>${getStatusLabel(a.status)}</td>
              </tr>
            `).join("")}
          </tbody>
        </table>
      </div>
      <div class="section">
        <div class="section-title">Appointment Details</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Selected Appointment:</span> ${appt.confirmation}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Location:</span> ${appt.location}</div>
        <div class="appt-summary-row"><span class="appt-summary-label">Notes:</span> ${appt.notes}</div>
        <div class="form-actions">
          <button class="btn danger" id="btn-cancel-appt">Cancel Appointment</button>
          <button class="btn secondary" id="btn-return-home">Return Home</button>
        </div>
      </div>
    `;
    // Row selection
    document.querySelectorAll('tbody tr').forEach(row => {
      row.onclick = () => {
        let idx = parseInt(row.dataset.idx,10);
        renderAppointments(idx);
      };
      row.onkeydown = e => {
        if (e.key === "Enter" || e.key === " ") {
          let idx = parseInt(row.dataset.idx,10);
          renderAppointments(idx);
        }
      };
    });
    // Cancel
    document.getElementById('btn-cancel-appt').onclick = () => {
      if (confirm("Are you sure you want to cancel this appointment?")) {
        appointments.splice(selectedAppointmentIdx,1);
        selectedAppointmentIdx = 0;
        renderAppointments(selectedAppointmentIdx);
      }
    };
    document.getElementById('btn-return-home').onclick = () => setScreen('home');
  }

  function renderHelp() {
    document.querySelector('main').innerHTML = `

      <div class="section-title">Frequently Asked Questions</div>
      <ul class="faq-list">
        ${FAQS.map(faq => `
          <li>

            <div class="faq-a">A: ${faq.a}</div>
          </li>
        `).join("")}
