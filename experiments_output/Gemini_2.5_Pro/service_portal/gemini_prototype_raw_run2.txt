    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // --- Mock Database ---
            let tickets = [
                {
                    id: '2025-104',
                    name: 'Sample User',
                    email: 'sample@university.edu',
                    category: 'Network Support',
                    description: 'Cannot connect to the campus Wi-Fi in the library.',
                    status: 'In Progress',
                    submittedDate: '2026-04-28',
                    lastUpdated: '2026-04-29',
                    technicianNotes: 'Issue currently being investigated. Checked router logs, seems to be a signal strength problem.'
                }
            ];
            let lastSubmittedTicketId = null;

            // --- DOM Elements ---
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-link');
            const mainNavLinks = document.querySelectorAll('#main-nav a');
            
            const submitForm = document.getElementById('submit-ticket-form');
            const confirmationDetails = document.getElementById('ticket-summary-details');
            const confirmTrackBtn = document.getElementById('confirm-track-btn');

            const searchInput = document.getElementById('search-ticket-id');
            const searchBtn = document.getElementById('search-btn');
            const trackResultsContainer = document.getElementById('track-request-results');
            const trackResultsDetails = document.getElementById('ticket-info-details');

            // --- Navigation Logic ---
            function showScreen(screenId) {
                screens.forEach(screen => {
                    screen.classList.remove('active');
                });
                const activeScreen = document.getElementById(`screen-${screenId}`);
                if (activeScreen) {
                    activeScreen.classList.add('active');
                }

                mainNavLinks.forEach(link => {
                    link.classList.remove('active');
                    if (link.getAttribute('href') === `#${screenId}`) {
                        link.classList.add('active');
                    }
                });
                window.scrollTo(0, 0);
            }

            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const screenId = link.getAttribute('href').substring(1);
                    showScreen(screenId);
                });
            });

            // --- Ticket Submission Logic ---
            submitForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const formData = new FormData(submitForm);
                const name = formData.get('name');
                const email = formData.get('email');
                const category = formData.get('category');
                const description = formData.get('description');

                if (!name || !email || !category || !description) {
                    alert('Please fill out all required fields.');
                    return;
                }

                const newTicket = {
                    id: `IT-${Math.floor(1000 + Math.random() * 9000)}`,
                    name: name,
                    email: email,
                    category: category,
                    description: description,
                    status: 'Open',
                    submittedDate: new Date().toISOString().split('T')[0],
                    lastUpdated: new Date().toISOString().split('T')[0],
                    technicianNotes: 'Ticket received and awaiting assignment.'
                };

                tickets.push(newTicket);
                lastSubmittedTicketId = newTicket.id;

                confirmationDetails.innerHTML = `
                    <p><strong>Ticket ID:</strong> ${newTicket.id}</p>
                    <p><strong>Category:</strong> ${newTicket.category}</p>
                    <p><strong>Submitted by:</strong> ${newTicket.name} (Student)</p>
                    <p><strong>Status:</strong> ${newTicket.status}</p>
                    <p><strong>Submitted Date:</strong> ${newTicket.submittedDate}</p>
                    <p><strong>Estimated Response Time:</strong> 24 Hours</p>
                `;
                
                submitForm.reset();
                showScreen('confirmation');
            });

            // --- Confirmation Screen Logic ---
            confirmTrackBtn.addEventListener('click', () => {
                if (lastSubmittedTicketId) {
                    searchInput.value = lastSubmittedTicketId;
                    performSearch();
                    showScreen('track');
                }
            });

            // --- Ticket Tracking Logic ---
            function performSearch() {
                const ticketId = searchInput.value.trim();
                trackResultsContainer.style.display = 'none';
                if (!ticketId) {
                    alert('Please enter a Ticket ID.');
                    return;
                }

                const foundTicket = tickets.find(t => t.id.toLowerCase() === ticketId.toLowerCase());

                if (foundTicket) {
                    trackResultsDetails.innerHTML = `
                        <p><strong>Ticket #:</strong> ${foundTicket.id}</p>
                        <p><strong>Category:</strong> ${foundTicket.category}</p>
                        <p><strong>Status:</strong> ${foundTicket.status}</p>
                        <p><strong>Submitted by:</strong> ${foundTicket.name} (Student)</p>
                        <p><strong>Last Updated:</strong> ${foundTicket.lastUpdated}</p>
                        <p><strong>Technician Notes:</strong></p>
                        <p>${foundTicket.technicianNotes}</p>
                    `;
                } else {
                    trackResultsDetails.innerHTML = `<p class="error">No ticket found with ID: ${ticketId}</p>`;
                }
                trackResultsContainer.style.display = 'block';
            }

            searchBtn.addEventListener('click', performSearch);
            searchInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    performSearch();
                }
            });

            // --- Initial State ---
            showScreen('home');
        });
    </script>

</body>
