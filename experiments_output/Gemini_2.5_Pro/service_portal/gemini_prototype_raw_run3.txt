    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-link, #main-nav a');
            const supportForm = document.getElementById('support-request-form');
            const searchBtn = document.getElementById('search-btn');
            const trackFromConfBtn = document.getElementById('track-from-conf-btn');

            let lastSubmittedTicket = {};

            // Mock database of tickets for tracking
            const mockTickets = {
                "2025-104": {
                    category: "Network Support",
                    status: "In Progress",
                    lastUpdated: "April 29, 2026",
                    notes: "Issue currently being investigated. Technician assigned."
                }
            };

            function showScreen(screenId) {
                screens.forEach(screen => {
                    screen.classList.remove('active');
                });
                const activeScreen = document.getElementById(screenId + '-screen');
                if (activeScreen) {
                    activeScreen.classList.add('active');
                }
            }

            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const targetScreen = e.target.getAttribute('data-target');
                    showScreen(targetScreen);
                });
            });

            supportForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const formData = new FormData(supportForm);
                const category = formData.get('issue-category');
                
                // Generate ticket details
                const ticketId = `IT-${Math.floor(1000 + Math.random() * 9000)}`;
                const submissionDate = new Date().toLocaleDateString('en-US', {
                    year: 'numeric', month: 'long', day: 'numeric'
                });

                // Store ticket info for tracking
                lastSubmittedTicket = {
                    id: ticketId,
                    category: category,
                    status: "Open",
                    date: submissionDate,
                    notes: "Ticket created. Awaiting technician assignment."
                };
                mockTickets[ticketId] = {
                    category: lastSubmittedTicket.category,
                    status: lastSubmittedTicket.status,
                    lastUpdated: lastSubmittedTicket.date,
                    notes: lastSubmittedTicket.notes
                };

                // Populate confirmation screen
                document.getElementById('conf-ticket-id').textContent = ticketId;
                document.getElementById('conf-category').textContent = category;
                document.getElementById('conf-status').textContent = 'Open';
                document.getElementById('conf-date').textContent = submissionDate;

                supportForm.reset();
                showScreen('confirmation');
            });

            function performSearch() {
                const ticketId = document.getElementById('ticket-id-search').value.trim();
                const infoContainer = document.getElementById('ticket-info-container');
                const notFoundMsg = document.getElementById('ticket-not-found');

                const ticket = mockTickets[ticketId];

                if (ticket) {
                    document.getElementById('track-ticket-num').textContent = ticketId;
                    document.getElementById('track-category').textContent = ticket.category;
                    document.getElementById('track-status').textContent = ticket.status;
                    document.getElementById('track-updated').textContent = ticket.lastUpdated;
                    document.getElementById('track-notes').textContent = `"${ticket.notes}"`;
                    
                    infoContainer.style.display = 'block';
                    notFoundMsg.style.display = 'none';
                } else {
                    infoContainer.style.display = 'none';
                    notFoundMsg.style.display = 'block';
                }
            }

            searchBtn.addEventListener('click', performSearch);
            
            document.getElementById('ticket-id-search').addEventListener('keypress', function (e) {
                if (e.key === 'Enter') {
                    performSearch();
                }
            });

            trackFromConfBtn.addEventListener('click', () => {
                document.getElementById('ticket-id-search').value = lastSubmittedTicket.id || '';
                showScreen('track');
                // Automatically perform search if an ID is present
                if (lastSubmittedTicket.id) {
                    performSearch();
                }
            });

            // Initialize with home screen
            showScreen('home');
        });
    </script>
