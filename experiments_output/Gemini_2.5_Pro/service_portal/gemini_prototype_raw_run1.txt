    <script>
        document.addEventListener('DOMContentLoaded', function () {
            // --- DATA STORE (In-memory simulation) ---
            const tickets = {
                "IT-104": {
                    id: "IT-104",
                    category: "Network Support",
                    status: "In Progress",
                    submittedBy: "Student",
                    lastUpdated: "April 29, 2026",
                    notes: "Issue currently being investigated.",
                    name: "Jane Doe",
                    email: "jane.d@university.edu",
                    description: "Wi-Fi is not connecting in the library.",
                    submittedDate: "April 28, 2026"
                }
            };

            // --- NAVIGATION ---
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-link');

            function showScreen(screenId) {
                screens.forEach(screen => {
                    screen.classList.remove('active');
                });
                const activeScreen = document.getElementById(screenId);
                if (activeScreen) {
                    activeScreen.classList.add('active');
                }
            }

            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const targetScreen = e.currentTarget.getAttribute('data-target');
                    showScreen(targetScreen);
                });
            });

            // --- TICKET SUBMISSION (Screen 2) ---
            const supportForm = document.getElementById('support-request-form');
            supportForm.addEventListener('submit', function (e) {
                e.preventDefault();
                
                const formData = new FormData(supportForm);
                const name = formData.get('name');
                const email = formData.get('email');
                const category = formData.get('category');
                const description = formData.get('description');

                // Generate a new ticket
                const ticketId = `IT-${Math.floor(1000 + Math.random() * 9000)}`;
                const submittedDate = new Date().toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' });
                
                const newTicket = {
                    id: ticketId,
                    category: category,
                    status: "Open",
                    submittedBy: "Student", // Assumption based on sketch
                    lastUpdated: submittedDate,
                    notes: "Ticket created.",
                    name: name,
                    email: email,
                    description: description,
                    submittedDate: submittedDate
                };
                tickets[ticketId] = newTicket;

                // Populate confirmation screen
                document.getElementById('conf-ticket-id').textContent = ticketId;
                document.getElementById('conf-category').textContent = category;
                document.getElementById('conf-submitted-by').textContent = `${name} (${email})`;
                document.getElementById('conf-date').textContent = submittedDate;

                // Store ticket ID for the "Track Request" button
                document.getElementById('track-from-confirmation-btn').dataset.ticketId = ticketId;

                supportForm.reset();
                showScreen('confirmation-screen');
            });

            // --- TICKET TRACKING (Screen 4) ---
            const searchBtn = document.getElementById('search-btn');
            const searchInput = document.getElementById('search-ticket-id');
            const ticketInfoContainer = document.getElementById('ticket-info-container');
            const searchError = document.getElementById('search-error');

            function performSearch(ticketId) {
                const ticket = tickets[ticketId.trim().toUpperCase()];
                
                ticketInfoContainer.style.display = 'none';
                searchError.style.display = 'none';

                if (ticket) {
                    document.getElementById('track-ticket-id').textContent = ticket.id;
                    document.getElementById('track-category').textContent = ticket.category;
                    const statusSpan = document.getElementById('track-status');
                    statusSpan.textContent = ticket.status;
                    statusSpan.className = 'status'; // Reset classes
                    if (ticket.status === 'Open') {
                        statusSpan.classList.add('status-open');
                    } else if (ticket.status === 'In Progress') {
                        statusSpan.classList.add('status-progress');
                    }
                    document.getElementById('track-submitted-by').textContent = ticket.submittedBy;
                    document.getElementById('track-last-updated').textContent = ticket.lastUpdated;
                    document.getElementById('track-notes').textContent = ticket.notes;
                    ticketInfoContainer.style.display = 'block';
                } else {
                    searchError.textContent = `Ticket with ID "${ticketId}" not found.`;
                    searchError.style.display = 'block';
                }
            }

            searchBtn.addEventListener('click', () => {
                const ticketId = searchInput.value;
                if (ticketId) {
                    performSearch(ticketId);
                }
            });

            // --- CONFIRMATION SCREEN to TRACK SCREEN ---
            document.getElementById('track-from-confirmation-btn').addEventListener('click', function(e) {
                const ticketId = e.currentTarget.dataset.ticketId;
                showScreen('track-screen');
                searchInput.value = ticketId;
                performSearch(ticketId);
            });

            // --- INITIAL STATE ---
            showScreen('home-screen');
        });
    </script>
