    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // --- STATE MANAGEMENT ---
            let appointments = [
                {
                    id: 'HC-2045',
                    patientName: 'Sarah Johnson',
                    type: 'General Check-up',
                    doctor: 'Dr. Smith',
                    date: '2025-10-15',
                    time: '10:30 AM',
                    status: 'Confirmed',
                    notes: 'Arrive 15 minutes early.'
                },
                {
                    id: 'HC-1988',
                    patientName: 'John Doe',
                    type: 'Follow-up Visit',
                    doctor: 'Dr. Lee',
                    date: '2025-10-22',
                    time: '01:00 PM',
                    status: 'Pending',
                    notes: 'Follow up on lab results.'
                }
            ];
            let selectedAppointmentId = null;

            // --- UI ELEMENTS ---
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-link');
            const scheduleForm = document.getElementById('schedule-form');
            const appointmentsList = document.getElementById('appointments-list');
            const detailsSection = document.getElementById('appointment-details-section');

            // --- NAVIGATION ---
            const showScreen = (screenId) => {
                screens.forEach(screen => {
                    screen.classList.toggle('active', screen.id === `screen-${screenId}`);
                });
                navLinks.forEach(link => {
                    link.classList.toggle('active', link.getAttribute('href') === `#${screenId}`);
                });
                window.scrollTo(0, 0);
            };

            document.getElementById('main-nav').addEventListener('click', (e) => {
                if (e.target.matches('.nav-link')) {
                    e.preventDefault();
                    const targetId = e.target.getAttribute('href').substring(1);
                    showScreen(targetId);
                }
            });

            document.getElementById('schedule-new-app-btn').addEventListener('click', () => showScreen('schedule'));
            
            document.querySelectorAll('.btn-link').forEach(link => {
                link.addEventListener('click', (e) => {
                    const targetId = e.target.getAttribute('data-target');
                    showScreen(targetId);
                });
            });

            // --- RENDERING ---
            const renderAppointments = () => {
                appointmentsList.innerHTML = '';
                if (appointments.length === 0) {
                    appointmentsList.innerHTML = '<tr><td colspan="4">No upcoming appointments.</td></tr>';
                } else {
                    appointments.forEach(app => {
                        const row = document.createElement('tr');
                        row.dataset.id = app.id;
                        row.innerHTML = `
                            <td>${app.id}</td>
                            <td>${app.doctor}</td>
                            <td>${new Date(app.date + 'T00:00:00').toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })}</td>
                            <td><span class="status status-${app.status.toLowerCase()}">${app.status}</span></td>
                        `;
                        appointmentsList.appendChild(row);
                    });
                }
                clearSelection();
            };

            const clearSelection = () => {
                selectedAppointmentId = null;
                detailsSection.style.display = 'none';
                document.querySelectorAll('#appointments-list tr').forEach(r => r.classList.remove('selected'));
            };

            // --- EVENT LISTENERS ---

            // Schedule Form Submission
            scheduleForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const newAppointment = {
                    id: 'HC-' + Math.floor(1000 + Math.random() * 9000),
                    patientName: document.getElementById('patient-name').value,
                    type: document.getElementById('appointment-type').value,
                    doctor: document.getElementById('preferred-doctor').value,
                    date: document.getElementById('appointment-date').value,
                    time: document.getElementById('appointment-time').value,
                    status: 'Confirmed',
                    notes: document.getElementById('notes').value || 'No additional notes.'
                };
                
                appointments.push(newAppointment);
                
                // Populate confirmation screen
                document.getElementById('conf-patient-name').textContent = newAppointment.patientName;
                document.getElementById('conf-type').textContent = newAppointment.type;
                document.getElementById('conf-doctor').textContent = newAppointment.doctor;
                const formattedDate = new Date(newAppointment.date + 'T00:00:00').toLocaleDateString('en-US', { month: 'long', day: 'numeric', year: 'numeric' });
                document.getElementById('conf-date').textContent = formattedDate;
                document.getElementById('conf-time').textContent = newAppointment.time;
                document.getElementById('conf-number').textContent = newAppointment.id;
                document.getElementById('conf-status').textContent = newAppointment.status;

                scheduleForm.reset();
                renderAppointments();
                showScreen('confirmation');
            });

            // Clear Schedule Form
            document.getElementById('clear-form-btn').addEventListener('click', () => {
                scheduleForm.reset();
            });

            // Select Appointment from List
            appointmentsList.addEventListener('click', (e) => {
                const row = e.target.closest('tr');
                if (!row || !row.dataset.id) return;

                clearSelection();
                selectedAppointmentId = row.dataset.id;
                row.classList.add('selected');

                const app = appointments.find(a => a.id === selectedAppointmentId);
                if (app) {
                    document.getElementById('details-conf-num').textContent = app.id;
                    document.getElementById('details-notes').textContent = app.notes;
                    detailsSection.style.display = 'block';
                }
            });

            // Cancel Appointment
            document.getElementById('cancel-app-btn').addEventListener('click', () => {
                if (selectedAppointmentId && confirm('Are you sure you want to cancel this appointment?')) {
                    appointments = appointments.filter(app => app.id !== selectedAppointmentId);
                    renderAppointments();
                    alert('Appointment cancelled successfully.');
                }
            });

            // --- INITIALIZATION ---
            renderAppointments();
            showScreen('home');
        });
    </script>
