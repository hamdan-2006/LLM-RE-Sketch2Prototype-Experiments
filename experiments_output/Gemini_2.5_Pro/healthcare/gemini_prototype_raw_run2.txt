    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // --- DATA STORE ---
            let appointments = [
                {
                    id: 'HC-2045',
                    patientName: 'Sarah Johnson',
                    email: 'sarah.j@example.com',
                    phone: '555-123-4567',
                    type: 'General checkup',
                    doctor: 'Dr. Smith',
                    date: '2025-10-15',
                    time: '10:30 AM',
                    notes: 'Arrive 15 minutes early.',
                    status: 'Confirmed'
                },
                {
                    id: 'HC-1988',
                    patientName: 'Sarah Johnson',
                    email: 'sarah.j@example.com',
                    phone: '555-123-4567',
                    type: 'Follow-up visits',
                    doctor: 'Dr. Lee',
                    date: '2025-10-28',
                    time: '03:00 PM',
                    notes: 'Follow up on test results.',
                    status: 'Pending'
                }
            ];
            let selectedAppointmentId = null;

            // --- UI ELEMENTS ---
            const mainNavLinks = document.querySelectorAll('.nav-link');
            const screens = document.querySelectorAll('main > section');
            const scheduleForm = document.getElementById('schedule-form');
            const appointmentsListBody = document.getElementById('appointments-list');
            const appointmentDetailsSection = document.getElementById('appointment-details-section');
            const appointmentDetailsContent = document.getElementById('appointment-details-content');
            const cancelBtn = document.getElementById('cancel-appointment-btn');

            // --- NAVIGATION ---
            const showScreen = (screenId) => {
                screens.forEach(screen => {
                    screen.classList.toggle('active', screen.id === screenId);
                });
                mainNavLinks.forEach(link => {
                    link.classList.toggle('active', link.hash === `#${screenId.replace('-screen', '')}`);
                });
                window.scrollTo(0, 0);
            };

            mainNavLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const targetScreenId = e.currentTarget.dataset.target || (e.currentTarget.hash.substring(1) + '-screen');
                    showScreen(targetScreenId);
                });
            });

            // --- APPOINTMENT RENDERING ---
            const renderAppointments = () => {
                appointmentsListBody.innerHTML = '';
                if (appointments.length === 0) {
                    appointmentsListBody.innerHTML = '<tr><td colspan="4">No upcoming appointments.</td></tr>';
                    return;
                }
                appointments.forEach(appt => {
                    const row = document.createElement('tr');
                    row.dataset.id = appt.id;
                    row.innerHTML = `
                        <td>${appt.id}</td>
                        <td>${appt.doctor}</td>
                        <td>${new Date(appt.date).toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric', timeZone: 'UTC' })}</td>
                        <td><span class="status status-${appt.status.toLowerCase()}">${appt.status}</span></td>
                    `;
                    row.addEventListener('click', () => handleSelectAppointment(appt.id));
                    appointmentsListBody.appendChild(row);
                });
            };

            const handleSelectAppointment = (id) => {
                selectedAppointmentId = id;
                const appt = appointments.find(a => a.id === id);
                if (!appt) return;

                document.querySelectorAll('#appointments-list tr').forEach(row => {
                    row.classList.toggle('selected', row.dataset.id === id);
                });

                appointmentDetailsContent.innerHTML = `
                    <p><strong>Confirmation #:</strong> ${appt.id}</p>
                    <p><strong>Patient Name:</strong> ${appt.patientName}</p>
                    <p><strong>Doctor:</strong> ${appt.doctor}</p>
                    <p><strong>Date & Time:</strong> ${new Date(appt.date).toLocaleDateString('en-US', { timeZone: 'UTC' })} at ${appt.time}</p>
                    <p><strong>Type:</strong> ${appt.type}</p>
                    <p><strong>Location:</strong> Community Health Clinic</p>
                    <p><strong>Notes:</strong> ${appt.notes || 'None'}</p>
                `;
                cancelBtn.disabled = appt.status === 'Cancelled';
                appointmentDetailsSection.classList.remove('hidden');
            };

            // --- FORM HANDLING ---
            scheduleForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const newAppointment = {
                    id: 'HC-' + Math.floor(1000 + Math.random() * 9000),
                    patientName: document.getElementById('patient-name').value,
                    email: document.getElementById('email').value,
                    phone: document.getElementById('phone').value,
                    type: document.getElementById('appointment-type').value,
                    doctor: document.getElementById('preferred-doctor').value,
                    date: document.getElementById('appointment-date').value,
                    time: document.getElementById('appointment-time').value,
                    notes: document.getElementById('notes').value,
                    status: 'Confirmed'
                };

                appointments.push(newAppointment);
                renderAppointments();
                
                const summaryContainer = document.getElementById('appointment-summary');
                summaryContainer.innerHTML = `
                    <p><strong>Patient Name:</strong> ${newAppointment.patientName}</p>
                    <p><strong>Appointment Type:</strong> ${newAppointment.type}</p>
                    <p><strong>Doctor:</strong> ${newAppointment.doctor}</p>
                    <p><strong>Date:</strong> ${new Date(newAppointment.date).toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric', timeZone: 'UTC' })}</p>
                    <p><strong>Time:</strong> ${newAppointment.time}</p>
                    <p><strong>Confirmation #:</strong> ${newAppointment.id}</p>
                    <p><strong>Status:</strong> <span class="status status-confirmed">${newAppointment.status}</span></p>
                `;

                scheduleForm.reset();
                showScreen('confirmation-screen');
            });

            // --- APPOINTMENT CANCELLATION ---
            cancelBtn.addEventListener('click', () => {
                if (!selectedAppointmentId) return;
                
                if (confirm('Are you sure you want to cancel this appointment?')) {
                    const apptIndex = appointments.findIndex(a => a.id === selectedAppointmentId);
                    if (apptIndex > -1) {
                        appointments[apptIndex].status = 'Cancelled';
                        renderAppointments();
                        appointmentDetailsSection.classList.add('hidden');
                        selectedAppointmentId = null;
                    }
                }
            });

            // --- INITIALIZATION ---
            const initialScreen = window.location.hash ? window.location.hash.substring(1) + '-screen' : 'home-screen';
            showScreen(initialScreen);
            renderAppointments();
        });
    </script>

</body>
