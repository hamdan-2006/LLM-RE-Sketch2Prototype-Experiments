    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // --- DATA STORE ---
            let appointments = [
                { id: 'HC-2045', doctor: 'Dr. Smith', date: '2025-10-15', time: '10:30 AM', type: 'General Checkup', status: 'Confirmed', notes: 'Arrive 15 minutes early.' },
                { id: 'HC-1988', doctor: 'Dr. Lee', date: '2025-10-28', time: '01:00 PM', type: 'Follow-up Visits', status: 'Pending', notes: 'Bring previous test results.' }
            ];
            let selectedAppointmentId = null;

            // --- DOM ELEMENTS ---
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-link');
            const appointmentForm = document.getElementById('appointment-form');
            const appointmentsListBody = document.getElementById('appointments-list');
            const appointmentDetailsDiv = document.getElementById('appointment-details');

            // --- FUNCTIONS ---
            const showScreen = (screenId) => {
                screens.forEach(screen => {
                    screen.classList.toggle('active', screen.id === screenId);
                });
                if (screenId === 'screen-my-appointments') {
                    renderAppointments();
                }
            };

            const renderAppointments = () => {
                appointmentsListBody.innerHTML = '';
                if (appointments.length === 0) {
                    appointmentsListBody.innerHTML = '<tr><td colspan="4">No upcoming appointments.</td></tr>';
                    appointmentDetailsDiv.innerHTML = '<p>No appointments to show.</p>';
                    return;
                }

                appointments.forEach(app => {
                    const row = document.createElement('tr');
                    row.dataset.id = app.id;
                    row.innerHTML = `
                        <td>${app.id}</td>
                        <td>${app.doctor}</td>
                        <td>${app.date}</td>
                        <td><span class="status-${app.status.toLowerCase()}">${app.status}</span></td>
                    `;
                    if (app.id === selectedAppointmentId) {
                        row.classList.add('selected');
                    }
                    appointmentsListBody.appendChild(row);
                });
                renderAppointmentDetails(selectedAppointmentId || appointments[0].id);
            };

            const renderAppointmentDetails = (appId) => {
                selectedAppointmentId = appId;
                const app = appointments.find(a => a.id === appId);

                document.querySelectorAll('#appointments-list tr').forEach(row => {
                    row.classList.toggle('selected', row.dataset.id === appId);
                });

                if (!app) {
                    appointmentDetailsDiv.innerHTML = '<p>Select an appointment from the list above to see details.</p>';
                    return;
                }

                appointmentDetailsDiv.innerHTML = `
                    <p><strong>Selected Appointment:</strong> ${app.id}</p>
                    <p><strong>Location:</strong> Community Health Clinic</p>
                    <p><strong>Notes:</strong> ${app.notes}</p>
                    <button class="btn btn-danger" id="cancel-btn" data-id="${app.id}">Cancel App</button>
                `;
            };

            const handleFormSubmit = (event) => {
                event.preventDefault();
                const newApp = {
                    id: 'HC-' + Math.floor(Math.random() * 9000 + 1000),
                    name: document.getElementById('patient-name').value,
                    type: document.getElementById('appointment-type').value,
                    doctor: document.getElementById('preferred-doctor').value,
                    date: document.getElementById('appointment-date').value,
                    time: document.getElementById('appointment-time').value,
                    notes: document.getElementById('notes').value || 'N/A',
                    status: 'Confirmed'
                };
                
                appointments.push(newApp);
                
                document.getElementById('conf-name').textContent = newApp.name;
                document.getElementById('conf-type').textContent = newApp.type;
                document.getElementById('conf-doctor').textContent = newApp.doctor;
                document.getElementById('conf-date').textContent = newApp.date;
                document.getElementById('conf-time').textContent = newApp.time;
                document.getElementById('conf-num').textContent = newApp.id;
                document.getElementById('conf-status').textContent = newApp.status;

                appointmentForm.reset();
                showScreen('screen-confirmation');
            };

            const handleCancelAppointment = (appId) => {
                if (confirm(`Are you sure you want to cancel appointment ${appId}?`)) {
                    appointments = appointments.filter(app => app.id !== appId);
                    selectedAppointmentId = null;
                    renderAppointments();
                }
            };

            // --- EVENT LISTENERS ---
            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    showScreen(e.target.dataset.target);
                });
            });

            appointmentForm.addEventListener('submit', handleFormSubmit);

            appointmentsListBody.addEventListener('click', (e) => {
                const row = e.target.closest('tr');
                if (row && row.dataset.id) {
                    renderAppointmentDetails(row.dataset.id);
                }
            });

            appointmentDetailsDiv.addEventListener('click', (e) => {
                if (e.target.id === 'cancel-btn') {
                    handleCancelAppointment(e.target.dataset.id);
                }
            });

            // --- INITIALIZATION ---
            renderAppointments();
            showScreen('screen-home');
        });
    </script>
