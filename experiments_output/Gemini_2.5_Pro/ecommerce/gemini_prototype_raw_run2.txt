    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const screens = document.querySelectorAll('.screen');
            const navLinks = document.querySelectorAll('.nav-bar a, button[data-target]');
            const TAX_RATE = 0.08;
            let cart = [];
            let lastOrderDetails = {};

            // --- NAVIGATION ---
            function showScreen(screenId) {
                screens.forEach(screen => {
                    screen.classList.remove('active');
                });
                const activeScreen = document.getElementById(screenId);
                if (activeScreen) {
                    activeScreen.classList.add('active');
                }
                window.scrollTo(0, 0);
            }

            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const targetId = e.currentTarget.getAttribute('data-target');
                    if (targetId === 'cart-page') {
                        renderCart();
                    }
                    showScreen(targetId);
                });
            });

            // --- CART LOGIC ---
            function addToCart(name, price) {
                const existingItem = cart.find(item => item.name === name);
                if (existingItem) {
                    // Per ambiguity note, quantity management is not supported.
                    // We will just alert the user.
                    alert(`${name} is already in your cart.`);
                } else {
                    cart.push({ name, price: parseFloat(price), quantity: 1 });
                    alert(`${name} has been added to your cart.`);
                }
            }

            function removeFromCart(index) {
                cart.splice(index, 1);
                renderCart();
            }

            function renderCart() {
                const cartContent = document.getElementById('cart-content');
                if (cart.length === 0) {
                    cartContent.innerHTML = '<p>Your cart is empty.</p><div class="cart-actions"><button class="btn btn-primary" data-target="products-page">Continue Shopping</button></div>';
                    // Re-attach listener for the new button
                    document.querySelector('#cart-content button[data-target]').addEventListener('click', (e) => showScreen(e.currentTarget.dataset.target));
                    return;
                }

                let subtotal = cart.reduce((acc, item) => acc + (item.price * item.quantity), 0);
                let tax = subtotal * TAX_RATE;
                let total = subtotal + tax;

                let tableHTML = `
                    <table class="cart-table">
                        <thead>
                            <tr>
                                <th>Product</th>
                                <th>Quantity</th>
                                <th>Price</th>
                                <th>Action</th>
                            </tr>
                        </thead>
                        <tbody>
                            ${cart.map((item, index) => `
                                <tr>
                                    <td>${item.name}</td>
                                    <td>${item.quantity}</td>
                                    <td>$${item.price.toFixed(2)}</td>
                                    <td><button class="btn btn-danger remove-item" data-index="${index}">Remove</button></td>
                                </tr>
                            `).join('')}
                        </tbody>
                    </table>
                    <div class="cart-totals">
                        <p>Subtotal: <strong>$${subtotal.toFixed(2)}</strong></p>
                        <p>Tax: <strong>$${tax.toFixed(2)}</strong></p>
                        <p>Total: <strong>$${total.toFixed(2)}</strong></p>
                    </div>
                    <div class="cart-actions">
                        <button class="btn btn-secondary" data-target="products-page">Continue Shopping</button>
                        <button class="btn btn-primary" id="proceed-to-checkout">Proceed to Checkout</button>
                    </div>
                `;
                cartContent.innerHTML = tableHTML;
                document.getElementById('checkout-total').textContent = `$${total.toFixed(2)}`;
            }

            // --- EVENT LISTENERS ---
            document.body.addEventListener('click', (e) => {
                if (e.target.classList.contains('add-to-cart')) {
                    const name = e.target.dataset.name;
                    const price = e.target.dataset.price;
                    addToCart(name, price);
                }
                if (e.target.classList.contains('remove-item')) {
                    const index = parseInt(e.target.dataset.index, 10);
                    removeFromCart(index);
                }
                if (e.target.id === 'proceed-to-checkout') {
                    showScreen('checkout-page');
                }
                if (e.target.matches('#cart-content button[data-target]')) {
                    showScreen(e.target.dataset.target);
                }
            });

            // --- CHECKOUT LOGIC ---
            document.getElementById('checkout-form').addEventListener('submit', (e) => {
                e.preventDefault();
                const form = e.target;
                if (form.checkValidity()) {
                    // Capture order details
                    const subtotal = cart.reduce((acc, item) => acc + (item.price * item.quantity), 0);
                    const tax = subtotal * TAX_RATE;
                    const total = subtotal + tax;
                    
                    lastOrderDetails = {
                        orderNumber: `ORD-${Date.now().toString().slice(-6)}`,
                        items: cart.map(item => item.name).join(', '),
                        totalPaid: total.toFixed(2),
                        paymentStatus: 'Approved',
                        estimatedDelivery: new Date(Date.now() + 5 * 24 * 60 * 60 * 1000).toLocaleDateString(),
                        shippingAddress: `${form.address.value}, ${form.city.value}, ${form.state.value} ${form.zip.value}`
                    };

                    renderOrderConfirmation();
                    showScreen('confirmation-page');
                    
                    // Reset cart and form
                    cart = [];
                    form.reset();
                } else {
                    alert('Please fill out all required fields.');
                }
            });

            function renderOrderConfirmation() {
                const summaryContainer = document.getElementById('order-summary-content');
                summaryContainer.innerHTML = `
                    <p><strong>Order #:</strong> ${lastOrderDetails.orderNumber}</p>
                    <p><strong>Items:</strong> ${lastOrderDetails.items}</p>
                    <p><strong>Total Paid:</strong> $${lastOrderDetails.totalPaid}</p>
                    <p><strong>Payment Status:</strong> ${lastOrderDetails.paymentStatus}</p>
                    <p><strong>Estimated Delivery:</strong> ${lastOrderDetails.estimatedDelivery}</p>
                    <p><strong>Shipping Address:</strong> ${lastOrderDetails.shippingAddress}</p>
                `;
            }

            // --- INITIALIZATION ---
            showScreen('home-page');
        });
    </script>
</body>
